    """
    –ù–∞–π—Ç–∏ —Å–æ—Ç—Ä—É–¥–Ω–∏–∫–∞ –ø–æ Telegram ID –ò–õ–ò username.

        city_ids = frozenset(int(c[0]) for c in city_rows)
        
        return StaffUser(
            id=staff.id,
            tg_id=staff.tg_user_id or tg_id,
            role=_map_staff_role(staff.role),
            is_active=bool(staff.is_active),
